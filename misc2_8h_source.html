37 inline Eigen::Matrix3d point_information_matrix(double distance)
38 {
39  Eigen::Matrix3d inf_mat = Eigen::Matrix3d::Identity();
40  /* Std dev of 1 pixel in xy boils down to identity for 0,0 and 1,1:
41  inf_mat(0,0) = 1.0/1.0;//-> 1/pixel_dist*pixel_dist
42  inf_mat(1,1) = 1.0/1.0;
43  */
44  inf_mat(2,2) = 1.0/depth_covariance(distance);
45 
46  return inf_mat;
